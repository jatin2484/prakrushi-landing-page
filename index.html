<script>

function toggleHam() {

const l = document.getElementById('n-box');

l.classList.toggle('active');

}


function shL(btn, id) {

document.querySelectorAll('.vault-nav a').forEach(a => a.classList.remove('active'));

document.querySelectorAll('.legal-content').forEach(c => c.classList.remove('active'));

btn.classList.add('active');

document.getElementById(id).classList.add('active');

}


// --- APP STORE COMPLIANCE ROUTER ---

// This logic captures the URL and opens the right section automatically.

function handleRoute(e, type) {

if (e) e.preventDefault();

const section = document.getElementById('contact');

section.scrollIntoView();

if (type === 'privacy') shL(document.getElementById('tab-privacy'), 'l2');

if (type === 'terms') shL(document.getElementById('tab-terms'), 'l1');

if (type === 'refund') shL(document.getElementById('tab-refund'), 'l3');

if (type === 'contact') shL(document.getElementById('tab-contact'), 'l4');

// Updates URL without reloading the page

window.history.pushState({}, "", "/" + type + "-policy");

}


// Check URL on load (e.g., if someone visits prakrushi.com/privacy-policy)

window.onload = function() {

const path = window.location.pathname;

if (path.includes('privacy')) handleRoute(null, 'privacy');

else if (path.includes('terms')) handleRoute(null, 'terms');

else if (path.includes('refund')) handleRoute(null, 'refund');

else if (path.includes('contact')) handleRoute(null, 'contact');

};

</script>
